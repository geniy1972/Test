Доступ к элементам

Запросы к DOM могут возвращать один элемент или несколько

Один элемент 
getElementById('one')
 
querySelector('li.hot')

пример

Несколько элементов 
getElementsByTagName('h1')
 
getElementsByTagName('li')
Несколько элементов 
getElementsByClassName('hot')
 
querySelectorAll('li[id]')

NodeList

  • Live NodeList (getElementsBy...)
  • Static NodeList (querySelector...)

var elements = document.getElementsByClassName('hot');
if (elements.length > 1) {
    var firstItem = elements.item(0);
}    
                            

var elements = document.getElementsByClassName('hot');
if (elements.length > 1) {
    var firstItem = elements[0];
}

Перебор элементов


var elements = document.querySelectorAll('li.hot');
if (elements.length > 0) {
    for (var i=0; i < elements.length; i++) {
        elements[i].className = 'cool';
    }
}
